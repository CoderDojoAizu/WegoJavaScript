JavaScript サンプルコード

□は半角スペース(空白文字)記号です。
1.JavaScript で作ってみよう
await commands.wego.bow();
await commands.wegoDance.SakuraIllumination();
2.変数を使ってみよう
let myName = "あるくメカトロウィーゴ";
await commands.wegoSpeech.setLanguage("ja-JP");
await commands.wegoSpeech.startText2Speech("僕の名前は" + myName);
3.条件分岐（if）を使ってみる
let value;
let fortune;
value = Math.floor(Math.random() * 3) + 1;
if (value == 1) {
  fortune = "最高";
} else if (value == 2) {
  fortune = "最悪";
} else {
  fortune = "普通";
}
await commands.wegoSpeech.setLanguage("ja-JP");
await commands.wegoSpeech.startText2Speech("今日の運勢は" + fortune);
4.関数を使ってみる
function addNumber(value1, value2) {
  return value1 + value2;
}
await commands.wegoSpeech.setLanguage("ja-JP");
await commands.wegoSpeech.startText2Speech(addNumber(10, 20));
5.ループを使ってみる
await commands.wegoSpeech.setLanguage("ja-JP");
for (let i = 1; i <= 3; i++) {
  await commands.wegoSpeech.startText2Speech("回数" + i);
}
6.一斉に踊ろう
let hasDanced = false;
async function checkTimeAndDance() {
  const now = new Date();
  const hours = now.getHours();
  const minutes = now.getMinutes();
  if (hours == 11 && minutes == 15 && !hasDanced) {
    console.log("時刻になりました。ダンスを開始します。");
    await commands.wegoDance.dance();
    hasDanced = true;
  }
}
setInterval(checkTimeAndDance, 100);
